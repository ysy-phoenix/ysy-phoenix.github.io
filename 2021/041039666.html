READ ME

1.本文整理了CTF中Crypto部分可能用到的一些函数，主要是古典密码。事实上绝大部分古典密码都有在线的解密网站，python中的一些库也有不少加/解密函数，以下函数均为为库函数中没有的。之所以做一个这样的整理，还是因为现在古典密码都是套娃加密，在涉及爆破时，在线网站无疑很费劲，因此有一个自己的库函数就显得很有必要。

2.本文的函数原理部分并无注释，但是绝大部分经过实践检验，如果出现错误，后续会更正（咕咕咕）。

3.有一些函数接口单一。举个栗子，my_base64_decode只接受字典，因此使用的时候需要将字符集改成字典的结构，当然也可以对源码进行魔改。

4.下文的凯撒加解密函数其实并不是通常意义下的，有相关需求只需对源代码进行简单修改即可。

5.以下除了仿射密码使用了gmpy2的invert函数，其余都不需要额外使用任何库。

def morseencode(decoded):
    """
    Morse cipher encryption function

    Example::

        encoded='.----/-..../...--/--.../-..'
        print(morsedecode(encoded))

    prints::

        1637D
    """
    morsestring = []
    decoded = decoded.upper()
    words = decoded.split(' ')
    for word in words:
        letters = list(word)
        morseword = []
        for letter in letters:
            morseletter = alphabet_to_morse[letter]
            morseword.append(morseletter)
        word = "/".join(morseword)
        morsestring.append(word)
    return " ".join(morsestring)


def morsedecode(encoded):
    """
    Morse cipher decryption function

    Example::

        decoded = '1637D'
        print(morseencode(decoded))

    prints::

        .----/-..../...--/--.../-..
    """
    characterstring = []
    words = encoded.split(" ")
    for word in words:
        letters = word.split("/")
        characterword = []
        for letter in letters:
            characterletter = morse_to_alphabet[letter]
            characterword.append(characterletter)
        word = "".join(characterword)
        characterstring.append(word)
    return " ".join(characterstring)


def my_base64_decode(encoded, dict):
    """
    Base64 decryption function, dict is the substitute dictionary

    Example::

        dict = {0: 'J', 1: 'K', 2: 'L', 3: 'M', 4: 'N', 5: 'O', 6: 'x', 7: 'y', 8: 'U', 9: 'V', 10: 'z', 11: 'A', 12: 'B',
        13: 'C', 14: 'D', 15: 'E', 16: 'F', 17: 'G', 18: 'H', 19: '7', 20: '8', 21: '9', 22: 'P', 23: 'Q', 24: 'I',
        25: 'a', 26: 'b', 27: 'c', 28: 'd', 29: 'e', 30: 'f', 31: 'g', 32: 'h', 33: 'i', 34: 'j', 35: 'k', 36: 'l',
        37: 'm', 38: 'W', 39: 'X', 40: 'Y', 41: 'Z', 42: '0', 43: '1', 44: '2', 45: '3', 46: '4', 47: '5', 48: '6',
        49: 'R', 50: 'S', 51: 'T', 52: 'n', 53: 'o', 54: 'p', 55: 'q', 56: 'r', 57: 's', 58: 't', 59: 'u', 60: 'v',
        61: 'w', 62: '+', 63: '/', 64: '='}
        print(morseencode(decoded))
    new_dict = {v : k for k, v in dict.items()}
    chipertext = 'FlZNfnF6Qol6e9w17WwQQoGYBQCgIkGTa9w3IQKw'
    print(my_base64_decode(chipertext, new_dict))
    prints::

        BJD{D0_Y0u_kNoW_Th1s_b4se_map}
    """
    res = ''
    p1 = len(encoded) // 4
    p2 = len(encoded) % 4
    for i in range(p1):
        str_p1 = encoded[i * 4] + encoded[i * 4 + 1] + encoded[i * 4 + 2] + encoded[i * 4 + 3]
        tmp_res = [bin(dict[x]).replace('0b', '').zfill(6) for x in str_p1]
        tmp_str = ''.join(tmp_res)
        tmp_res = [tmp_str[x: x + 8] for x in [0, 8, 16]]
        for j in range(3):
            res += chr(int(tmp_res[j], 2))
    if p2:
        str_p2 = encoded[4 * p1:]
        tmp_res = [bin(dict[x]).replace('0b', '').zfill(6) for x in str_p2]
        tmp_str = ''.join(tmp_res)
        tmp_res = [tmp_str[x: x + 8] for x in [0, 8, 16]]
        for j in range(p2):
            res += chr(int(tmp_res[j], 2))
    return res


def shift_encrypt(m, k):
    """
    Shift cipher encryption function

    Example::

        m = "flag{easy_easy_crypto}"
        k = "3124"
        c = shift_encrypt(m, k)
        print(c)

    prints::

        lafgea{s_eyay_scyprt}o
    """
    l = len(k)
    c = ""
    for i in range(0, len(m), l):
        tmp_c = [""] * l
        if i + l > len(m):
            tmp_m = m[i:]
        else:
            tmp_m = m[i:i + l]
        for kindex in range(len(tmp_m)):
            tmp_c[int(k[kindex]) - 1] = tmp_m[kindex]
        c += "".join(tmp_c)
    return c


def shift_decrypt(c, k):
    """
    Shift cipher decryption function

    Example::

        c = 'lafgea{s_eyay_scyprt}o'
        k = "3124"
        print(shift_decrypt(c, k))

    prints::

        flag{easy_easy_crypto}
    """
    l = len(k)
    m = ""
    for i in range(0, len(c), l):
        tmp_m = [""] * l
        if i + l > len(c):
            tmp_c = c[i:]
            use = []
            for kindex in range(len(tmp_c)):
                use.append(int(k[kindex]) - 1)
            use.sort()
            for kindex in range(len(tmp_c)):
                tmp_m[kindex] = tmp_c[use.index(int(k[kindex]) - 1)]
        else:
            tmp_c = c[i:i + l]
            for kindex in range(len(tmp_c)):
                tmp_m[kindex] = tmp_c[int(k[kindex]) - 1]
        m += "".join(tmp_m)
    return m


def c01248_decode(c):
    """
    01248(yunying) cipher decryption function

    Example::

        c = "8842101220480224404014224202480122"
        print(c01248_decode(c))

    prints::

        welldone
    """
    l = c.split("0")
    origin = "abcdefghijklmnopqrstuvwxyz"
    r = ""
    for i in l:
        tmp = 0
        for num in i:
            tmp += int(num)
        r += origin[tmp - 1]
    return r


def zhalan_encrypt(m, k):
    """
    zhalan cipher encryption function

    Example::

        m = "flag{zhalan_mima_hahaha}"
        k = 4
        c = zhalan_encrypt(m, k)
        print(c)

    prints::

        f{lm_alzaihhahnmaaga_ah}
    """
    chip = []
    for i in range(0, len(m), k):
        if i + k >= len(m):
            tmp_m = m[i:]
        else:
            tmp_m = m[i:i + k]
        chip.append(tmp_m)
    c = ""
    for i in range(k):
        for tmp_m in chip:
            if i < len(tmp_m):
                c += tmp_m[i]
    return c


def zhalan_decrypt(c, k):
    """
    zhalan cipher decryption function

    Example::

        c = "f{lm_alzaihhahnmaaga_ah}"
        k = 4
        m = zhalan_encrypt(m, k)
        print(m)

    prints::

        flag{zhalan_mima_hahaha}
    """

    l = len(c)
    partnum = l // k
    if l % k != 0:
        partnum += 1
    m = [""] * l
    for i in range(0, l, partnum):
        if i + partnum >= len(c):
            tmp_c = c[i:]
        else:
            tmp_c = c[i:i + partnum]
        for j in range(len(tmp_c)):
            m[j * k + i // partnum] = tmp_c[j]
    return "".join(m)


def variation_zhalan_decode(string, key):
    """
    a variation of zhalan cipher decryption function

    Example::

        string = "NxC_GWj}e{1ohUJn$i0i1?p@oiHz?"
        key = 3
        m = variation_zhalan_decode(string, key)
        print(m)

    prints::

        Nep{x1@oChoU_JinG$HiW0zij1??}
    """
    de_key = 2 * key - 2
    length = len(string) // de_key
    r = len(string) % de_key
    delist = []
    for i in range(key):
        delist.append('')
    # 确定第一个分组
    if r == 0:
        delist[0] += string[0:length]
        s = length
    else:
        delist[0] += string[0:length + 1]
        s = length + 1
    for i in range(1, key - 1):
        l = length * 2
        if r > i:
            l += 1
        if r > de_key - i:
            l += 1

        delist[i] += string[s:s + l]
        s = s + l
    delist[key - 1] += string[s:]
    destr = ''
    j = 0
    for i in range(0, len(string)):
        destr += delist[j][0]
        delist[j] = delist[j][1:]
        if j == key - 1:
            flag = 0
        if j == 0:
            flag = 1
        if flag:
            j += 1
        else:
            j -= 1
    return destr


def caesar_encrypt(m, k):
    """
    Caesar cipher encryption function

    Example::

        m = "flag{kaisamima}"
        k = 3
        c = caesar_encrypt(m, k)
        print(c)

    prints::

        iodj~ndlvdplpd
    """
    r = ""
    for i in m:
        r += chr((ord(i) + k) % 128)
    return r


def caesar_decrypt(c, k):
    """
    Caesar cipher decryption function

    Example::

        c = "iodj~ndlvdplpd"
        k = 3
        m = caesar_decrypt(c, k)
        print(m)

    prints::

        iodj~ndlvdplpd
    """
    r = ""
    for i in c:
        r += chr((ord(i) - k) % 128)
    return r


def caesar_brute(c, match_str):
    """
    Caesar brute function

    Example::

        c = "39.4H/?BA2,0.2@.?J"
        print(caesar_brute(c, "flag{"))

    prints::

        i['flag{kaisamima}']
    """
    result = []
    for k in range(128):
        tmp = caesar_decrypt(c, k)
        if match_str in tmp:
            result.append(tmp)
    return result


def rot13(m):
    """
    rot13 decryption/encryption function

    Example::

        c = "2cf24dba5fb0a30e26e83b2ac5b9e29e1b161e5c1fa7425e73043362938b9824"
        print(rot13(c))
        print(rot13(rot13(c)))
    prints::

        2ps24qon5so0n30r26r83o2np5o9r29r1o161r5p1sn7425r73043362938o9824
        2cf24dba5fb0a30e26e83b2ac5b9e29e1b161e5c1fa7425e73043362938b9824
    """
    r = ""
    for i in m:
        if ord(i) in range(ord('A'), ord('Z') + 1):
            r += chr((ord(i) + 13 - ord('A')) % 26 + ord('A'))
        elif ord(i) in range(ord('a'), ord('z') + 1):
            r += chr((ord(i) + 13 - ord('a')) % 26 + ord('a'))
        else:
            r += i
    return r


def atabsh_encode(m):
    """
    atabsh encryption/decryption function

    Example::

        m = "flag{ok_atabsh_flag}"
        c = atabsh_encode(m)
        print(c)
        print(atabsh_encode(c))

    prints::

        flag{ok_atabsh_flag}
        uozt{lp_zgzyhs_uozt}
    """
    alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    Origin = alphabet + alphabet.lower()
    TH_A = alphabet[::-1]
    TH_a = alphabet.lower()[::-1]
    TH = TH_A + TH_a
    r = ""
    for i in m:
        tmp = Origin.find(i)
        if tmp != -1:
            r += TH[tmp]
        else:
            r += i
    return r


def substitution_encode(m, k, origin="abcdefghijklmnopqrstuvwxyz"):
    """
    substitution encryption function

    Example::

        m = "flag{good_good_study}"
        k = "zugxjitlrkywdhfbnvosepmacq"
        c = substitution_encode(m, k)
        print(c)

    prints::

        iwzt{tffx_tffx_osexc}
    """
    r = ""
    for i in m:
        if origin.find(i) != -1:
            r += k[origin.find(i)]
        else:
            r += i
    return r

def affine_encode(m, a, b, origin="abcdefghijklmnopqrstuvwxyz"):
    """
    affine encryption function

    Example::

        m = "affinecipher"
        a = 5
        b = 8
        c = affine_encode(m, a, b)
        print(c)

    prints::

        ihhwvcswfrcp
    """
    r = ""
    for i in m:
        if origin.find(i) != -1:
            r += origin[(a * origin.index(i) + b) % len(origin)]
        else:
            r += i
    return r


def affine_decode(c, a, b, origin="abcdefghijklmnopqrstuvwxyz"):
    """
    affine decryption function

    Example::

        c = "ihhwvcswfrcp"
        a = 5
        b = 8
        m = affine_decode(c, a, b)
        print(c)

    prints::

        affinecipher
    """
    r = ""
    n = len(origin)
    ai = invert(a, n)
    for i in c:
        if origin.find(i) != -1:
            r += origin[(ai * (origin.index(i) - b)) % len(origin)]
        else:
            r += i
    return r


def affine_guessab(m1, c1, m2, c2, origin="abcdefghijklmnopqrstuvwxyz"):
    """
    affine brute function -> guess a and b

    Example::

        print(affine_guessab('a', 'i', 'f', 'h'))

    prints::

        (mpz(5), mpz(8))
    """
    x1 = origin.index(m1)
    x2 = origin.index(m2)
    y1 = origin.index(c1)
    y2 = origin.index(c2)
    n = len(origin)
    dxi = invert(abs(x1 - x2), n)
    if x1 - x2 < 0:
        dxi *= -1
    a = dxi * (y1 - y2) % n
    b = (y1 - a * x1) % n
    return (a, b)


def gen_cheese_map(k, use_Q=True, upper=True):
    """
    get chess function

    Example::

        print(gen_cheese_map("helloword"))

    prints::

        ['HELOW', 'RDABC', 'FGIKM', 'NPQST', 'UVXYZ']
    """
    k = k.upper()
    k0 = ""
    origin = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    for i in k:
        if i not in k0:
            k0 += i
    for i in origin:
        if i not in k0:
            k0 += i
    if use_Q == True:
        k0 = k0[0:k0.index("J")] + k0[k0.index("J") + 1:]
    else:
        k0 = k0[0:k0.index("Q")] + k0[k0.index("Q") + 1:]
    if upper == False:
        k0 = k0.lower()
    assert len(k0) == 25
    r = []
    for i in range(5):
        r.append(k0[i * 5:i * 5 + 5])
    return r


def _playfair_2char(tmp, map):
    """
    find char of playfair map while encoding

    """
    for i in range(5):
        for j in range(5):
            if map[i][j] == tmp[0]:
                ai = i
                aj = j
            if map[i][j] == tmp[1]:
                bi = i
                bj = j
    if ai == bi:
        axi = ai
        bxi = bi
        axj = (aj + 1) % 5
        bxj = (bj + 1) % 5
    elif aj == bj:
        axj = aj
        bxj = bj
        axi = (ai + 1) % 5
        bxi = (bi + 1) % 5
    else:
        axi = ai
        axj = bj
        bxi = bi
        bxj = aj
    return map[axi][axj] + map[bxi][bxj]


def playfair_encode(m, k="", cheese_map=[]):
    """
    playfair encryption function

    Example::

        m = "Hide the gold in the tree stump"
        k = "playfairexample"
        c = playfair_encode(m, k)
        print(c)

    prints::

        ['BM', 'OD', 'ZB', 'XD', 'NA', 'BE', 'KU', 'DM', 'UI', 'XM', 'MO', 'UV', 'IF']
    """
    m = m.upper()
    origin = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    tmp = ""
    for i in m:
        if i in origin:
            tmp += i
    m = tmp
    assert k != "" or cheese_map != []
    if cheese_map == []:
        map = gen_cheese_map(k)
    else:
        map = cheese_map
    m0 = []
    idx = 0
    while idx < len(m):
        tmp = m[idx:idx + 2]
        if tmp[0] != tmp[1]:
            m0.append(tmp)
            idx += 2
        elif tmp[0] != "X":
            m0.append(tmp[0] + "X")
            idx += 1
        else:
            m0.append(tmp[0] + "Q")
            idx += 1
        if idx == len(m) - 1:
            if tmp[0] != "X":
                m0.append(tmp[0] + "X")
                idx += 1
            else:
                m0.append(tmp[0] + "Q")
    r = []
    for i in m0:
        r.append(_playfair_2char(i, map))
    return r


def _playfair_2char_decode(tmp, map):
    """
    find char of playfair map while decoding

    """

    for i in range(5):
        for j in range(5):
            if map[i][j] == tmp[0]:
                ai = i
                aj = j
            if map[i][j] == tmp[1]:
                bi = i
                bj = j
    if ai == bi:
        axi = ai
        bxi = bi
        axj = (aj - 1) % 5
        bxj = (bj - 1) % 5
    elif aj == bj:
        axj = aj
        bxj = bj
        axi = (ai - 1) % 5
        bxi = (bi - 1) % 5
    else:
        axi = ai
        axj = bj
        bxi = bi
        bxj = aj
    return map[axi][axj] + map[bxi][bxj]


def playfair_decode(c, k="", cheese_map=[]):
    """
    playfair decryption function

    Example::

        k = "playfairexample"
        c = ['BM', 'OD', 'ZB', 'XD', 'NA', 'BE', 'KU', 'DM', 'UI', 'XM', 'MO', 'UV', 'IF']
        print(playfair_decode(c, k))

    prints::

        HIDETHEGOLDINTHETREXESTUMP
    """
    assert k != "" or cheese_map != []
    if cheese_map == []:
        map = gen_cheese_map(k)
    else:
        map = cheese_map
    r = []
    for i in c:
        r.append(_playfair_2char_decode(i, map))
    return "".join(r)


def polybius_encode(m, k="", name="ADFGX", cheese_map=[]):
    """
    polybius encryption function

    Example::

        m = "helloworld"
        k = "abcd"
        c = polybius_encode(m, k)
        print(c)

    prints::

        ['DF', 'AX', 'FA', 'FA', 'FG', 'XD', 'FG', 'GD', 'FA', 'AG']
    """
    m = m.upper()
    assert k != "" or cheese_map != []
    if cheese_map == []:
        map = gen_cheese_map(k)
    else:
        map = cheese_map
    r = []
    for x in m:
        for i in range(5):
            for j in range(5):
                if map[i][j] == x:
                    r.append(name[i] + name[j])
    return r


def polybius_decode(c, k="", name="ADFGX", cheese_map=[]):
    """
    polybius decryption function

    Example::

        k = "abcd"
        c = ['DF', 'AX', 'FA', 'FA', 'FG', 'XD', 'FG', 'GD', 'FA', 'AG']
        print(polybius_decode(c, k))

    prints::

        HELLOWORLD
    """
    assert k != "" or cheese_map != []
    if cheese_map == []:
        map = gen_cheese_map(k)
    else:
        map = cheese_map
    r = ""
    for x in c:
        i = name.index(x[0])
        j = name.index(x[1])
        r += map[i][j]
    return r
